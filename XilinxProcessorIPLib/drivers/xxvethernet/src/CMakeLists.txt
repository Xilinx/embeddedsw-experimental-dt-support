cmake_minimum_required(VERSION 2.8.9)
include(${CMAKE_CURRENT_SOURCE_DIR}/../../../../cmake/common.cmake NO_POLICY_SCOPE)
project(xxvethernet)

collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_SOURCES xxxvethernet.c)
collect (PROJECT_LIB_HEADERS xxxvethernet.h)
collect (PROJECT_LIB_SOURCES xxxvethernet_g.c)
collect (PROJECT_LIB_HEADERS xxxvethernet_hw.h)
collect (PROJECT_LIB_SOURCES xxxvethernet_sinit.c)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(xxvethernet STATIC ${_sources})
set_target_properties(xxvethernet PROPERTIES LINKER_LANGUAGE C)
install(TARGETS xxvethernet LIBRARY DESTINATION ${CMAKE_SOURCE_DIR}/build ARCHIVE DESTINATION ${CMAKE_INSTALL_LIBDIR})
