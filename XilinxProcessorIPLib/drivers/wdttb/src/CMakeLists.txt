cmake_minimum_required(VERSION 2.8.9)
include(${CMAKE_CURRENT_SOURCE_DIR}/../../../../cmake/common.cmake NO_POLICY_SCOPE)
project(wdttb)

collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_SOURCES xwdttb.c)
collect (PROJECT_LIB_HEADERS xwdttb.h)
collect (PROJECT_LIB_SOURCES xwdttb_g.c)
collect (PROJECT_LIB_HEADERS xwdttb_hw.h)
collect (PROJECT_LIB_HEADERS xwdttb_l.h)
collect (PROJECT_LIB_SOURCES xwdttb_selftest.c)
collect (PROJECT_LIB_SOURCES xwdttb_sinit.c)
collect (PROJECT_LIB_SOURCES xwdttb_config.c)
collect (PROJECT_LIB_HEADERS xwdttb_config.h)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(wdttb STATIC ${_sources})
set_target_properties(wdttb PROPERTIES LINKER_LANGUAGE C)
install(TARGETS wdttb LIBRARY DESTINATION ${CMAKE_SOURCE_DIR}/build ARCHIVE DESTINATION ${CMAKE_INSTALL_LIBDIR})
