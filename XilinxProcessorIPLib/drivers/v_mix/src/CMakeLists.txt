cmake_minimum_required(VERSION 2.8.9)
include(${CMAKE_CURRENT_SOURCE_DIR}/../../../../cmake/common.cmake NO_POLICY_SCOPE)
project(v_mix)

collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
include_directories(${CMAKE_BINARY_DIR}/include)
collect (PROJECT_LIB_SOURCES xv_mix_sinit.c)
collect (PROJECT_LIB_SOURCES xv_mix_l2_intr.c)
collect (PROJECT_LIB_SOURCES xv_mix_l2.c)
collect (PROJECT_LIB_HEADERS xv_mix_l2.h)
collect (PROJECT_LIB_HEADERS xv_mix.h)
collect (PROJECT_LIB_SOURCES xv_mix_linux.c)
collect (PROJECT_LIB_SOURCES xv_mix_g.c)
collect (PROJECT_LIB_HEADERS xv_mix_hw.h)
collect (PROJECT_LIB_SOURCES xv_mix.c)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
add_library(v_mix STATIC ${_sources})
set_target_properties(v_mix PROPERTIES LINKER_LANGUAGE C)
install(TARGETS v_mix LIBRARY DESTINATION ${CMAKE_SOURCE_DIR}/build ARCHIVE DESTINATION ${CMAKE_INSTALL_LIBDIR})
